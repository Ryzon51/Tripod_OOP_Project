package model;
import java.time.LocalDate;

public class HarvestLot extends FarmItem {
    private String status;
    private Double pricePerUnit;

    public HarvestLot(String n, double q, String u, LocalDate d, String nt, String s, Double p) {
        // Init fields...
        this.status = s; this.pricePerUnit = p;
    }
    @Override public String getItemType() { return "Harvest"; }
    @Override public String toCSVString() { 
        return String.format("%d,%s,%f,%s,%s", getId(), getName(), getQuantity(), status, pricePerUnit); 
    }
    public String getStatus() { return status; }
    public Double getPricePerUnit() { return pricePerUnit; }
}
